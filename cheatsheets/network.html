üîç Host Discovery

Network Discovery Scan 
# Nmap host discovery
nmap -sn 192.168.1.0/24                  # Ping sweep
nmap -sn -PS22,80,443 192.168.1.0/24     # TCP SYN ping
nmap -sn -PA80,443 192.168.1.0/24        # TCP ACK ping
nmap -sn -PU53 192.168.1.0/24            # UDP ping

# Netdiscover (ARP)
netdiscover -r 192.168.1.0/24
netdiscover -i eth0 -r 192.168.1.0/24

# ARP scan
arp-scan -l
arp-scan 192.168.1.0/24

# fping
fping -a -g 192.168.1.0/24 2>/dev/null

# Masscan - Fast scanning
masscan 192.168.1.0/24 -p1-65535 --rate=1000

# Responder for passive discovery
responder -I eth0 -A  # Analyze mode only

üì° Nmap Scanning

Nmap Scan Types Nmap 
# Basic scans
nmap -sT 192.168.1.1               # TCP connect
nmap -sS 192.168.1.1               # TCP SYN (stealth)
nmap -sU 192.168.1.1               # UDP scan
nmap -sV 192.168.1.1               # Version detection
nmap -sC 192.168.1.1               # Default scripts
nmap -O 192.168.1.1                # OS detection
nmap -A 192.168.1.1                # Aggressive (OS, version, scripts, traceroute)

# Port specification
nmap -p 22,80,443 192.168.1.1      # Specific ports
nmap -p 1-1000 192.168.1.1         # Port range
nmap -p- 192.168.1.1               # All 65535 ports
nmap --top-ports 100 192.168.1.1
